from broker.oanda_candles import OandaCandleFetcher
from engine.broker_chart import BrokerChart
from config.settings import OANDA_API_KEY, OANDA_ACCOUNT_ID, PAIRS

import matplotlib.pyplot as plt


def choose_timeframe():
    print("\nAvailable timeframes:")
    print("M1  M5  M15  M30  H1  H4  D")
    tf = input("Enter timeframe: ").upper()
    if tf not in ["M1", "M5", "M15", "M30", "H1", "H4", "D"]:
        tf = "H1"
    return tf


def main():
    timeframe = choose_timeframe()

    print("\n🚀 Broker Chart Mirror")
    print(f"Pairs     : {PAIRS}")
    print(f"Timeframe : {timeframe}")
    print("Mode      : Chart Only\n")

    fetcher = OandaCandleFetcher(
        OANDA_ACCOUNT_ID,
        OANDA_API_KEY,
        practice=True
    )

    charts = []
    for pair in PAIRS:
        charts.append(
            BrokerChart(pair, fetcher, timeframe)
        )

    # matplotlib owns the main thread
    plt.show()


if __name__ == "__main__":
    main()
